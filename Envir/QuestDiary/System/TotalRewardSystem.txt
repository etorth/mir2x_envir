[@TotalRewardSystem]
{
#ACT
Mov A8 ""
FormatStr "FLD_NAME='%s' and FLD_COND1=%s and " %A0 %D0

[@TotalRewardSystem_Setting_Pre()]
#ACT
Mov A3 %ARG(1)

#ACT
Mov A5 %ARG(1)
Mov D5 %ARG(2)
Mov D6 %ARG(3)
#IF
!Equal D6 0
Checkfame
#ACT
inc P9 %D6
SetFame %P9

ISADMIN
#ACT

#ACT

#ACT

#ACT

break

#ACT

!Equal A3 "ExpMap"

#ACT

#ACT
